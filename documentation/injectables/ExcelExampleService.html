  private title = "Listado de Alumnos";

  constructor() { }

  generateStudentsListExcel(studentsList: Array<any>) {
    this.workbook = new Workbook();

    let worksheet = this.workbook.addWorksheet("Listado");

    worksheet.columns = [
      { key: 'nombre', width: 20 },
      { key: 'edad', width: 10 },
      { key: 'genero', width: 10 },
    ];

    worksheet.addRow([this.title]);
    worksheet.mergeCells('A1', 'C1');
    worksheet.addRow(["Nombre", "Edad", "GÃ©nero"]);

    worksheet.getRow(1).font = { bold: true, size: 20 };
    worksheet.getRow(1).alignment = { vertical: 'middle', horizontal: 'center' };
    worksheet.getRow(2).font = { bold: true };
    worksheet.getRow(2).alignment = { vertical: 'middle', horizontal: 'center' };
    worksheet.getRow(2).fill = {
      type: 'pattern',
      pattern: 'solid',
      fgColor: { argb: 'F08080' },
    };

    worksheet.getColumn("B").alignment = { vertical: 'middle', horizontal: 'center' };
    worksheet.getColumn("C").alignment = { vertical: 'middle', horizontal: 'center' };

    studentsList.forEach((item) => {
      let cols = [];

      cols.push(item.nombre);
      cols.push(item.edad);
      cols.push(item.genero);

      worksheet.addRow(cols);
    });

    worksheet.getCell('B9').value = { formula: 'SUM(B3:B8)', result: 7 };

    const imageId2 = this.workbook.addImage({
      base64: this.base64Img,
      extension: 'png',
    });

    worksheet.addImage(imageId2, 'F1:H8');

    worksheet.addImage(imageId2, {
      tl: { col: 0, row: 10 },
      ext: { width: 70, height: 100 }
    });
  }

  getExcel(studentsList: Array<any>) {
    this.generateStudentsListExcel(studentsList);

    this.workbook.xlsx.writeBuffer().then((data) => {
      let blob = new Blob([data], { type: FILE_TYPE });
      FileSaver.saveAs(blob, this.getFileName())
    });
